\section{Một số hướng nghiên cứu}

Chương này trình bày bài báo khoa học gần đây liên quan tới giảm chiều dữ liệu. Kiến thức nền về NCE và NEG đã trình bày ở Chương~2 (Mục~\ref{sec:nce-neg}); các phương pháp giảm chiều (PCA, KPCA, t-SNE, UMAP, \ldots) ở Chương~3 (Mục~\ref{sec:extension-viz}). Chương này \emph{không} lặp lại các định nghĩa đó mà tham chiếu trực tiếp.

\subsection{Thống nhất t-SNE và UMAP qua học tương phản}
\label{sec:research-contrastive}

Bài báo \emph{From t-SNE to UMAP with contrastive learning} của Damrich, Böhm, Hamprecht và Kobak~\cite{damrich2023contrastive} (ICLR 2023) đặt ra câu hỏi: \textbf{mối quan hệ toán học chính xác giữa t-SNE và UMAP là gì?} Hai phương pháp được xây dựng từ nền tảng dường như khác nhau---KL divergence (MLE) vs.\ fuzzy cross-entropy (negative sampling)---nhưng bài báo chỉ ra rằng chiếc cầu nối là \emph{học tương phản} (contrastive learning).

\subsubsection{Khung nhúng láng giềng tương phản}
\label{sec:neighbor-embedding-framework}

\paragraph{Thiết lập chung.}
Cho $n$ điểm $\mathbf{x}_1,\ldots,\mathbf{x}_n \in \R^D$ và tọa độ nhúng $\theta = \{\mathbf{y}_1,\ldots,\mathbf{y}_n\} \subset \R^d$. Định nghĩa \emph{không gian cặp}:
\begin{equation}
  \mathcal{X} = \{(i,j) : 1 \le i < j \le n\}, \quad |\mathcal{X}| = \tbinom{n}{2}.
  \label{eq:pair-space}
\end{equation}
Từ đồ thị $k$-láng giềng đối xứng (skNN), xây dựng phân phối dữ liệu $p$ trên $\mathcal{X}$: $p(ij) > 0$ trên các cạnh skNN (chuẩn hóa $\sum_{ij} p(ij) = 1$) và $p(ij) = 0$ với các cặp còn lại. Nhiễu $\xi$ là phân phối đồng đều trên $\mathcal{X}$.

Kernel hạ chiều $\phi : \R_{\ge 0} \to (0,1]$ biến khoảng cách $d_{ij} = \|\mathbf{y}_i - \mathbf{y}_j\|$ thành ``xác suất tương đồng''. Phân phối mô hình:
\begin{equation}
  q_{\theta,Z}(ij) = \frac{\phi(d_{ij})}{Z}, \quad
  Z(\theta) = \sum_{(k,l)\in\mathcal{X}} \phi(d_{kl}).
  \label{eq:model-dist}
\end{equation}

\textbf{Phương pháp} & \textbf{Kernel $\phi(d)$} & \textbf{Hằng số $Z$} & \textbf{Khung tối ưu} & \textbf{Nghiệm $q^*$} \\
\midrule
t-SNE        & $\dfrac{1}{1+d^2}$ & $Z(\theta)$ (tính đủ)  & MLE & $p$ \\[4pt]

UMAP (hiệu dụng) & $\dfrac{1}{d^2}$ & $\dfrac{\binom{n}{2}}{m}$ (cố định) & NEG & $\overline{Z}\,p$ \\

\subsubsection{Bước 1 --- t-SNE là MLE; NC-t-SNE xấp xỉ t-SNE bằng NCE}
\label{sec:mle-to-nce}

\paragraph{t-SNE $=$ MLE.}
Hàm loss t-SNE là cross-entropy (tương đương MLE) giữa $p$ và $q_{\theta,Z(\theta)}$:
\begin{equation}
  \mathcal{L}_{\text{t-SNE}}(\theta)
  = -\sum_{(i,j)\in\mathcal{X}} p(ij)\,\log q_{\theta,Z(\theta)}(ij)
  = -\sum_{ij} p(ij)\log\frac{\phi(d_{ij})}{Z(\theta)}.
  \label{eq:tsne-loss-ch4}
\end{equation}
Mỗi bước gradient đòi hỏi tính $Z(\theta) = \sum_{kl}\phi(d_{kl})$ trên toàn bộ $\binom{n}{2}$ cặp---chi phí $\mathcal{O}(n^2)$ mỗi epoch.

\paragraph{NC-t-SNE $=$ NCE cho bài toán nhúng.}
Áp dụng NCE (Mục~\ref{sec:nce-def}) với không gian $\mathcal{X}$, phân phối $p$, nhiễu $\xi$ đồng đều, và mô hình $q_{\theta,Z}$ từ~\eqref{eq:model-dist} (với $Z$ là \emph{tham số học được}). Thay vào loss NCE~\eqref{eq:nce-loss}:
\begin{equation}
  \mathcal{L}_{\text{NC-t-SNE}}(\theta, Z)
  = -\E_{ij \sim p}\log\dfrac{\dfrac{\phi(d_{ij})}{Z}}{\dfrac{\phi(d_{ij})}{Z} + m\xi(ij)}
  \;-\; m\,\E_{ij \sim \xi}\log\dfrac{m\xi(ij)}{\dfrac{\phi(d_{ij})}{Z} + m\xi(ij)}.
  \label{eq:nc-tsne-loss}
\end{equation}
Theo Định lý~\ref{thm:nce}, mọi cực tiểu $(\theta^*, Z^*)$ thỏa $q_{\theta^*,Z^*} = p$, tức $Z^*$ hội tụ đúng về $Z(\theta^*)$. \textbf{Kết luận:} NC-t-SNE có cùng nghiệm tối ưu như t-SNE nhưng \emph{không} phải tính $Z(\theta)$ trực tiếp.

\subsubsection{Bước 2 --- Cố định $Z$: từ NCE đến NEG}
\label{sec:nce-to-neg}

Bước từ NCE sang NEG---cố định $Z \leftarrow \overline{Z}$ thay vì để $Z$ tự học---là bước \emph{duy nhất} làm thay đổi nghiệm tối ưu, và là chìa khóa giải thích sự khác biệt hình thái giữa t-SNE và UMAP.

\begin{theorem}[Hệ quả 2 --- Damrich et al., 2023]
\label{thm:cor2}
Cho $\overline{Z},\, m > 0$. Giả sử $\xi(x) > 0\ \forall x \in \mathcal{X}$ và tồn tại $\theta^*$ sao cho $q_{\theta^*} = \overline{Z}\,p$. Khi đó $\theta^*$ là cực tiểu (và mọi cực trị) của:
\begin{equation}
  \mathcal{L}_{\overline{Z}}(\theta)
  = -\E_{x \sim p}\log\frac{q_\theta(x)}{q_\theta(x) + \overline{Z}\,m\xi(x)}
  - m\,\E_{x \sim \xi}\log\frac{\overline{Z}\,m\xi(x)}{q_\theta(x) + \overline{Z}\,m\xi(x)}.
  \label{eq:gen-nce-loss}
\end{equation}
\end{theorem}

\begin{proof}
Đặt $\tilde{q}_\theta(x) := \frac{q_\theta(x)}{\overline{Z}}$. Chia cả tử và mẫu trong mỗi logarit của~\eqref{eq:gen-nce-loss} cho $\overline{Z}$:
\begin{align*}
  \frac{q_\theta(x)}{q_\theta(x) + \overline{Z}\,m\xi(x)}
  &= \frac{\frac{q_\theta(x)}{\overline{Z}}}{\frac{q_\theta(x)}{\overline{Z}} + m\xi(x)}
  = \frac{\tilde{q}_\theta(x)}{\tilde{q}_\theta(x) + m\xi(x)},\\[4pt]
  \frac{\overline{Z}\,m\xi(x)}{q_\theta(x) + \overline{Z}\,m\xi(x)}
  &= \frac{m\xi(x)}{\frac{q_\theta(x)}{\overline{Z}} + m\xi(x)}
  = \frac{m\xi(x)}{\tilde{q}_\theta(x) + m\xi(x)}.
\end{align*}
Do đó $\mathcal{L}_{\overline{Z}}(\theta) \equiv \mathcal{L}_{\text{NCE}}[\tilde{q}_\theta]$---đúng dạng loss NCE chuẩn~\eqref{eq:nce-loss} áp dụng cho mô hình $\tilde{q}_\theta$. Theo Định lý~\ref{thm:nce}, cực tiểu đạt khi $\tilde{q}_{\theta^*} = p$, tức $q_{\theta^*} = \overline{Z}\,p$.
\end{proof}

\paragraph{NEG là trường hợp riêng $\overline{Z} = \frac{|\mathcal{X}|}{m}$.}
Với $\xi$ đồng đều ($\xi(x) = \frac{1}{|\mathcal{X}|}$), hệ số nhiễu trong~\eqref{eq:gen-nce-loss} rút gọn:
\[
  \overline{Z}\,m\xi(x) = \frac{|\mathcal{X}|}{m}\cdot m \cdot \frac{1}{|\mathcal{X}|} = 1,
\]
và mẫu số trở thành $q_\theta(x) + 1$---đúng dạng loss NEG~\eqref{eq:neg-loss} (với $\phi_\theta \leftarrow q_\theta$). Theo Định lý~\ref{thm:cor2}, nghiệm NEG thỏa $q_{\theta^*} = \overline{Z}\,p$ với $\overline{Z} = \frac{|\mathcal{X}|}{m}$ rất lớn.

\begin{center}
\fbox{\parbox{0.85\textwidth}{%
\textbf{NCE} ($Z$ học được): \quad $q_{\theta^*} = p$ \hfill (phân phối đúng).\\[4pt]
\textbf{NEG} ($\overline{Z} = \frac{|\mathcal{X}|}{m}$ cố định): \quad $q_{\theta^*} = \overline{Z}\,p$ \hfill (phóng đại $\overline{Z}$ lần).%
}}
\end{center}

\subsubsection{Bước 3 --- UMAP thực chất là NEG}
\label{sec:umap-is-neg}

\begin{lemma}[Bổ đề 3 --- Damrich et al., 2023]
\label{lem:umap-neg}
Loss hiệu dụng của UMAP (sau khi tính đến cơ chế lấy mẫu âm) chính là loss NEG~\eqref{eq:neg-loss} với kernel $\tilde{q}_\theta(ij) = \frac{1}{d_{ij}^2}$.
\end{lemma}

\begin{proof}
Böhm et al.~\cite{bohm2022attraction} chỉ ra rằng loss hiệu dụng UMAP có dạng:
\begin{equation}
  \mathcal{L}_{\text{UMAP}}(\theta)
  = -\E_{ij \sim p}\log\phi(d_{ij})
  - m\,\E_{ij \sim \xi}\log\!\bigl(1 - \phi(d_{ij})\bigr),
  \label{eq:umap-eff-ch4}
\end{equation}
với $\phi(d) = \frac{1}{1+d^2}$ và $m = 5$ (mặc định). Đặt $\tilde{q}(ij) := \frac{1}{d_{ij}^2}$. Khi đó:
\begin{equation}
  \phi(d_{ij})
  = \frac{1}{1+d_{ij}^2}
  = \frac{d_{ij}^{-2}}{1+d_{ij}^{-2}}
  = \frac{\tilde{q}(ij)}{\tilde{q}(ij)+1}, \qquad
  1-\phi(d_{ij}) = \frac{1}{\tilde{q}(ij)+1}.
  \label{eq:umap-neg-equiv}
\end{equation}
Thay vào~\eqref{eq:umap-eff-ch4}:
\[
  \mathcal{L}_{\text{UMAP}}(\theta)
  = -\E_{p}\log\frac{\tilde{q}(ij)}{\tilde{q}(ij)+1}
  - m\,\E_{\xi}\log\frac{1}{\tilde{q}(ij)+1},
\]
đúng bằng loss NEG~\eqref{eq:neg-loss} với $\phi_\theta \leftarrow \tilde{q}_\theta$.
\end{proof}

Áp dụng Định lý~\ref{thm:cor2}: UMAP dùng NEG với hằng số cố định
\begin{equation}
  \overline{Z}_{\text{UMAP}} = \frac{|\mathcal{X}|}{m} = \dfrac{\binom{n}{2}}{m}
  = \frac{n(n-1)}{2m} = \mathcal{O}(n^2),
  \label{eq:z-umap}
\end{equation}
trong khi $Z(\theta)$ của t-SNE $\approx 50n$--$100n$ \cite{bohm2022attraction}---nhỏ hơn $\overline{Z}_{\text{UMAP}}$ bậc $n$.

Bước NCE$\to$NEG (cố định $\overline{Z}$) là bước \emph{duy nhất} thay đổi nghiệm tối ưu. UMAP khác Neg-t-SNE chỉ ở kernel ($\frac{1}{d^2}$ vs.\ Cauchy), nhưng với $d_{ij}$ lớn cả hai xấp xỉ nhau ($\phi(d) \approx \frac{1}{d^2}$) nên sự khác biệt kernel gần như không ảnh hưởng hình thái nhúng.

\subsubsection{Phổ nhúng: $\overline{Z}$ điều khiển tỷ lệ lực đẩy/hút}
\label{sec:contrastive-gradients}

\paragraph{Cấu trúc gradient.}
Với Cauchy kernel $\phi(d) = \frac{1}{1+d^2}$ và $\mathbf{r}_{ij} = \mathbf{y}_i - \mathbf{y}_j$, ta có $\frac{\partial\phi(ij)}{\partial\mathbf{y}_i} = -2\phi(ij)^2\,\mathbf{r}_{ij}$ (khai triển tường minh tại Phụ lục~\ref{app:nce-gradient}). Ba gradient của MLE, NCE, NEG có cấu trúc song song, chỉ khác ở hệ số chuẩn hóa và hệ số \emph{dampening}:
\begin{align}
  \frac{\partial\mathcal{L}_{\text{t-SNE}}}{\partial\mathbf{y}_i}
  &= 2\sum_{j \neq i}\!\left(\frac{p(ij)}{q_\theta(ij)} - 1\right)\frac{1}{Z(\theta)}
    \cdot\frac{\partial\phi(ij)}{\partial\mathbf{y}_i},
  \label{eq:grad-tsne}\\[6pt]
  \frac{\partial\mathcal{L}_{\text{NC-t-SNE}}}{\partial\mathbf{y}_i}
  &= 2\sum_{j \neq i}\underbrace{\dfrac{m\xi(ij)}{\dfrac{\phi(ij)}{Z} + m\xi(ij)}}_{\xrightarrow{m\to\infty} 1}
    \!\left(\frac{p(ij)}{q_{\theta,Z}(ij)} - 1\right)\frac{1}{Z}
    \cdot\frac{\partial\phi(ij)}{\partial\mathbf{y}_i},
  \label{eq:grad-nctsne}\\[6pt]
  \frac{\partial\mathcal{L}_{\text{Neg-t-SNE}}}{\partial\mathbf{y}_i}
  &= 2\sum_{j \neq i}\underbrace{\dfrac{m\xi(ij)}{\dfrac{\phi(ij)}{\overline{Z}} + m\xi(ij)}}_{\in[0.5,\,1]}
    \!\left(\dfrac{p(ij)}{\dfrac{\phi(ij)}{\overline{Z}}} - 1\right)\frac{1}{\overline{Z}}
    \cdot\frac{\partial\phi(ij)}{\partial\mathbf{y}_i}.
  \label{eq:grad-negtsne}
\end{align}
Ba gradient chỉ khác ở: (i)~hệ số chuẩn hóa ($Z(\theta)$, $Z$, hay $\overline{Z}$), và (ii)~hệ số dampening (bằng $1$ cho MLE; hội tụ về $1$ khi $m\to\infty$ cho NCE; nằm trong $[0.5,1]$ cho NEG).

\paragraph{Tách lực hút và lực đẩy.}
Xét gradient Neg-t-SNE~\eqref{eq:grad-negtsne}. Tổng gồm hai loại cặp:
\begin{itemize}
  \item \textbf{Cạnh skNN} ($p(ij) > 0$): số hạng $\dfrac{p(ij)}{\frac{\phi(ij)}{\overline{Z}}}$ chiếm ưu thế
    $\Rightarrow$ \textbf{lực hút}, kéo $\mathbf{y}_i$ về phía $\mathbf{y}_j$.
    Lực hút \emph{không phụ thuộc} $\overline{Z}$: $\dfrac{p(ij)\overline{Z}}{\phi(ij)} \cdot \dfrac{1}{\overline{Z}} = \dfrac{p(ij)}{\phi(ij)}$.

  \item \textbf{Cặp nhiễu} ($p(ij) = 0$): còn lại $-\dfrac{\phi(ij)}{\overline{Z}}$
    $\Rightarrow$ \textbf{lực đẩy}, đẩy $\mathbf{y}_i$ ra xa $\mathbf{y}_j$. Lực đẩy tỷ lệ \emph{nghịch} với $\overline{Z}$:
\end{itemize}
\begin{equation}
  F_{\text{rep}} \propto \frac{\phi(ij)^2}{\overline{Z}} \propto \frac{1}{\overline{Z}}.
  \label{eq:repulsion-scaling}
\end{equation}
(Chi tiết khai triển và chứng minh tại Phụ lục~\ref{app:nce-gradient}.)

\paragraph{Phổ nhúng từ t-SNE đến UMAP.}
Phương trình~\eqref{eq:repulsion-scaling} giải thích trực tiếp sự khác biệt hình thái:
\begin{itemize}
  \item $\overline{Z} = Z(\theta) \approx 50n$--$100n$ \textbf{(t-SNE)}: lực đẩy mạnh, không gian nhúng dàn rộng, cụm phân tán rõ---mô hình ưu tiên bảo toàn \emph{cấu trúc cục bộ chi tiết}.
  \item $\overline{Z} = \dfrac{\binom{n}{2}}{m} = \mathcal{O}(n^2)$ \textbf{(UMAP)}: lực đẩy yếu hơn bậc $n$, lực hút thống trị $\Rightarrow$ \emph{cụm siêu co chặt}.
  \item $\overline{Z} \gg \dfrac{\binom{n}{2}}{m}$: lực đẩy gần triệt tiêu, các cụm bắt đầu sáp nhập.
\end{itemize}

\begin{remark}
$\overline{Z}$ là \emph{tham số trượt} (slider) cho phép nội suy liên tục giữa hành vi t-SNE và UMAP. Bằng cách đặt $\overline{Z} = Z_{\text{lo}}\cdot\exp\!\bigl(s(\log Z_{\text{hi}} - \log Z_{\text{lo}})\bigr)$ với $s \in [0,1]$, $Z_{\text{lo}} = Z_{\text{t-SNE}}$, $Z_{\text{hi}} = \dfrac{\binom{n}{2}}{m}$, ta thu được họ phương pháp gọi là \textbf{Neg-t-SNE spectrum}.
\end{remark}

\paragraph{Ổn định số trị: Cauchy vs.\ nghịch bình phương.}
Neg-t-SNE dùng Cauchy kernel $\phi(d) = \frac{1}{1+d^2} \in (0,1]$ (chặn trên), nên lực đẩy tối đa là $\log 2$ dù $d_{ij} \to 0$. Ngược lại, UMAP dùng $\tilde{q}(d) = \frac{1}{d^2} \to \infty$ khi $d \to 0$, gây hiệu ứng ``catapult'' và bất ổn gradient---lý do UMAP cần \textit{learning rate annealing} trong thực hành. Đây là ưu điểm thực tiễn của Neg-t-SNE.

\textbf{Hình thái cụm}         & Dàn trải     & Dàn trải      & Linh hoạt     & Siêu co chặt \\
\textbf{Ổn định số}            & Ổn định      & Ổn định       & Ổn định       & Cần annealing \\
\bottomrule
\end{tabular}
\end{table}

\subsection{Cân bằng cục bộ--toàn cục trong trực quan hóa giảm chiều}
\label{sec:research-pacmap}

Bài báo \emph{Understanding How Dimension Reduction Tools Work: An Empirical Approach to Deciphering t-SNE, UMAP, TriMap, and PaCMAP for Data Visualization} của Wang, Huang, Rudin và Shaposhnik~\cite{wang2021pacmap} (JMLR 2021) đặt ra câu hỏi: \textbf{khi nhúng dữ liệu chiều cao xuống hai chiều, thành phần nào của hàm mất mát quyết định việc giữ cấu trúc cục bộ hay toàn cục?} Bài báo phân tích thực nghiệm lực hút--lực đẩy trên t-SNE, UMAP và TriMap, rồi đề xuất PaCMAP (\textit{Pairwise Controlled Manifold Approximation Projection})---thuật toán thiết kế trực tiếp ba loại cặp điểm và lịch trọng số theo giai đoạn thay vì chỉ tinh chỉnh siêu tham số trên đồ thị kNN.

\subsubsection{Bài toán trực quan hóa}
\label{sec:pacmap-problem}

Cho $n$ điểm $\mathbf{x}_1,\ldots,\mathbf{x}_n \in \R^D$ và embedding $\theta = \{\mathbf{y}_1,\ldots,\mathbf{y}_n\} \subset \R^2$. Khi $D \gg 2$, không thể bảo toàn đồng thời mọi khoảng cách; scatter plot hai chiều dễ bị diễn giải sai nếu người xem coi khoảng cách giữa cụm hay kích thước cụm như cấu trúc thật của dữ liệu gốc. t-SNE và UMAP thường làm nổi bật láng giềng cục bộ nhưng có thể biến dạng bố cục toàn cục; ngược lại, phương pháp quá ưu tiên cấu trúc toàn cục có thể làm mờ các cụm nhỏ. Wang et al.~\cite{wang2021pacmap} muốn hiểu thành phần nào của loss, đồ thị và khởi tạo gây ra sự khác biệt đó, rồi thiết kế thuật toán cân bằng hai mục tiêu tốt hơn.

\subsubsection{Rainbow figure và sáu nguyên tắc thiết kế loss}
\label{sec:pacmap-forces}

Các thuật toán nhúng láng giềng có thể diễn giải qua lực hút trên cặp cần gần nhau và lực đẩy trên cặp không nên chồng lên nhau. Wang et al.~\cite{wang2021pacmap} vẽ \emph{rainbow figure}---biểu diễn độ lớn gradient theo khoảng cách trong không gian gốc và không gian nhúng---để so sánh t-SNE, UMAP và TriMap thay vì chỉ đối chiếu công thức. Từ phân tích này, tác giả rút ra sáu nguyên tắc cho hàm mất mát giữ cục bộ tốt:
\begin{enumerate}
  \item Láng giềng bị đặt xa trong embedding phải chịu lực hút \emph{đủ mạnh}.
  \item Điểm không lân cận đã ở xa không nên tiếp tục chi phối quá mức quá trình tối ưu.
  \item Lực đẩy giữa điểm xa phải đủ để tránh chồng lấn, nhưng không được triệt tiêu lực hút cục bộ.
  \item Chỉ thu hút láng giềng gần nhất và đẩy điểm xa \emph{chưa đủ} để phục hồi cấu trúc toàn cục.
  \item Cặp ở khoảng cách trung gian mang thông tin quan trọng về bố cục tổng thể.
  \item Kết quả embedding phụ thuộc cả hàm mất mát \emph{và} phép khởi tạo (TriMap nhạy cảm với khởi tạo PCA).
\end{enumerate}

\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Phương pháp} & \textbf{Thông tin ưu tiên} & \textbf{Nhận xét trong bài báo} \\
\midrule
t-SNE  & Láng giềng cục bộ & Cụm rõ; bố cục toàn cục không nên diễn giải trực tiếp. \\[2pt]
UMAP   & Đồ thị lân cận mờ & Lực đẩy yếu hơn t-SNE; cụm co chặt hơn. \\[2pt]
TriMap & Quan hệ bộ ba & Giữ toàn cục tốt hơn nếu khởi tạo PCA phù hợp. \\[2pt]

\bottomrule
\end{tabular}
\end{center}

\subsubsection{Bước 1 --- Ba loại cặp điểm}
\label{sec:pacmap-pairs}

PaCMAP xây dựng ba tập cặp trên không gian gốc $\R^D$:
\begin{itemize}
  \item \textbf{Neighbor pairs} $\mathcal{N}$: mỗi điểm $i$ nối với $n_{\text{NB}}$ láng giềng gần nhất (mặc định $n_{\text{NB}} = 10$), dùng khoảng cách chuẩn hóa theo độ lệch chuẩn địa phương.
  \item \textbf{Mid-near pairs} $\mathcal{M}$: với mỗi $i$, lấy mẫu sáu điểm ngẫu nhiên và ghép $i$ với điểm \emph{gần thứ hai} trong sáu điểm đó; $|\mathcal{M}| = n_{\text{MN}} = \text{MN\_ratio} \cdot n_{\text{NB}}$ (mặc định $\text{MN\_ratio} = 0.5$).
  \item \textbf{Further pairs} $\mathcal{F}$: ghép $i$ với điểm lấy mẫu ngẫu nhiên \emph{không} thuộc láng giềng; $|\mathcal{F}| = n_{\text{FP}} = \text{FP\_ratio} \cdot n_{\text{NB}}$ (mặc định $\text{FP\_ratio} = 2$).
\end{itemize}
Thay vì tăng số láng giềng để giữ toàn cục, PaCMAP dùng cặp trung gian $\mathcal{M}$ với lực hút \emph{yếu}: kéo các cụm gần nhau về phía nhau mà không phá cấu trúc cục bộ do $\mathcal{N}$ duy trì.

\subsubsection{Bước 2 --- Hàm loss và lịch trọng số ba giai đoạn}
\label{sec:pacmap-loss}

Khoảng cách bình phương trong không gian nhúng (có dịch chuyển $+1$ để tránh chia cho không):
\begin{equation}
  \tilde{d}_{ab} = \|\mathbf{y}_a - \mathbf{y}_b\|^2 + 1, \quad a,b \in \{1,\ldots,n\}.
  \label{eq:pacmap-dist}
\end{equation}
Hàm loss PaCMAP~\cite{wang2021pacmap}:
\begin{equation}
  \mathcal{L}_{\text{PaCMAP}}(\theta)
  = w_{\text{NB}} \!\!\sum_{(i,j)\in\mathcal{N}} \frac{\tilde{d}_{ij}}{C_{\text{Med}} + \tilde{d}_{ij}}
  + w_{\text{MN}} \!\!\sum_{(i,k)\in\mathcal{M}} \frac{\tilde{d}_{ik}}{C_{\text{Lg}} + \tilde{d}_{ik}}
  + w_{\text{FP}} \!\!\sum_{(i,l)\in\mathcal{F}} \frac{1}{1 + \tilde{d}_{il}},
  \label{eq:pacmap-loss}
\end{equation}
với $C_{\text{Med}} = 10$ (lực hút mạnh trên $\mathcal{N}$) và $C_{\text{Lg}} = 10\,000$ (lực hút yếu trên $\mathcal{M}$). Số hạng thứ ba là lực đẩy trên $\mathcal{F}$. Hệ số $C_{\text{Lg}} \gg C_{\text{Med}}$ đảm bảo mid-near chỉ tác động nhẹ so với neighbor.

PaCMAP tối ưu~\eqref{eq:pacmap-loss} theo ba giai đoạn (Adam, mặc định 450 iteration):
\begin{enumerate}
  \item \textbf{Giai đoạn 1} (100 iter): $w_{\text{NB}} = 2$, $w_{\text{FP}} = 1$, $w_{\text{MN}}(t)$ giảm tuyến tính từ $1000$ xuống $3$---dựng bố cục toàn cục trước.
  \item \textbf{Giai đoạn 2} (250 iter): $w_{\text{MN}}$ tiếp tục giảm, cân bằng dần ba loại cặp.
  \item \textbf{Giai đoạn 3} (100 iter): giảm vai trò mid-near, tinh chỉnh quan hệ láng giềng cục bộ.
\end{enumerate}

\subsection{Nhận xét tổng quan: xu hướng, hạn chế và câu hỏi mở}
\label{sec:research-trends}

\paragraph{Xu hướng 1: Thống nhất dưới khung toán học chung.}
Damrich et al.~\cite{damrich2023contrastive} chứng minh t-SNE, NC-t-SNE, Neg-t-SNE và UMAP là các trường hợp của NCE/NEG trên đồ thị skNN; trước đó Böhm et al.~\cite{bohm2022attraction} đã mô tả phổ hút--đẩy trong t-SNE. Xu hướng này giúp cộng đồng \emph{hiểu} thay vì chỉ \emph{dùng} các thuật toán DR.

\paragraph{Xu hướng 2: Thiết kế mục tiêu lai cho trực quan hóa.}
Wang et al.~\cite{wang2021pacmap} cho thấy nghiên cứu DR trực quan hóa chuyển từ tiêu chuẩn đơn (phương sai PCA, geodesic Isomap) sang thiết kế lai: cụm cục bộ phải rõ nhưng bố cục toàn cục không được tùy tiện. PaCMAP minh họa bằng cặp trung gian và lịch trọng số thay vì chỉ tinh chỉnh perplexity hay $k$.

\paragraph{Xu hướng 3: Liên hệ với học biểu diễn tự giám sát.}
NC-t-SNE, Neg-t-SNE và UMAP gần với SimCLR~\cite{damrich2023contrastive}: khác biệt chính là cách xây dựng cặp tương đồng (đồ thị kNN cố định vs.\ data augmentation), mở khả năng chuyển giao kỹ thuật giữa hai lĩnh vực.

\paragraph{Hạn chế và câu hỏi mở.}
\begin{itemize}
  \item \textbf{Giả định mô hình:} Định lý~\ref{thm:nce} và Định lý~\ref{thm:cor2} giả định mô hình khớp được $p$; trong thực tế phân phối trên skNN có nhiều vị trí bằng 0 mà kernel luôn dương không khớp hoàn hảo.
  \item \textbf{Chọn $\overline{Z}$ hay lịch trọng số:} Neg-t-SNE cần đặt $\overline{Z}$ thủ công; PaCMAP dùng lịch $w_{\text{MN}}$ cố định---câu hỏi mở là liệu có thể \emph{học} các hệ số này từ dữ liệu.
  \item \textbf{Đánh giá đa thước đo:} Trustworthiness và KNN chỉ đo cục bộ; cần thêm triplet accuracy hay continuity để đánh giá toàn cục như Wang et al.
  \item \textbf{DR trực tuyến:} Cả hai khung đều giả định toàn bộ dữ liệu có sẵn; mở rộng cho streaming là hướng tiềm năng.
\end{itemize}
