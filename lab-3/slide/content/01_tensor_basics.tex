\vspace{0.2cm}
\[
    \mathcal{X}
    \approx
    \sum_{r=1}^{R}
    \mathbf{a}_r^{(1)}\circ\mathbf{a}_r^{(2)}\circ\mathbf{a}_r^{(3)},
    \qquad
    x_{ijk}\approx\sum_{r=1}^{R}A^{(1)}_{ir}A^{(2)}_{jr}A^{(3)}_{kr}
\]
\begin{itemize}
    \item $R$ là số component/rank; mỗi component là một pattern đồng thời trên các mode.
\end{itemize}

\[
    \mathcal{X}\approx
    \mathcal{G}\times_1 U^{(1)}\times_2 U^{(2)}\times_3 U^{(3)},

    x_{ijk}\approx\sum_{p,q,r}G_{pqr}U^{(1)}_{ip}U^{(2)}_{jq}U^{(3)}_{kr}
\]
\begin{itemize}
    \item Core $\mathcal{G}$ mô tả tương tác giữa latent dimensions; Tucker trở về CP khi core là superdiagonal.
\end{itemize}

\begin{frame}{Rank, uniqueness và độ khó}
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
        \textbf{Kruskal uniqueness}
        \[
            k_A+k_B+k_C\ge 2R+2
        \]
        \begin{itemize}
            \item CP duy nhất đến hoán vị và co giãn.
            \item Đây là lý do CP hữu ích cho diễn giải pattern.
        \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
        \textbf{Khác với ma trận}
        \begin{itemize}
            \item Matrix rank tính được hiệu quả bằng SVD.
            \item CP rank của tensor bậc $N\ge3$ là NP-hard.
            \item Truncation tối ưu như Eckart-Young không còn đúng tổng quát.
        \end{itemize}
    \end{column}
\end{columns}
\end{frame}
