\section{Tensor trong Học sâu}

\begin{frame}{Tensor xuất hiện tự nhiên trong học sâu}
\begin{itemize}
    \item Batch ảnh:
    \[
        X\in\mathbb{R}^{B\times C\times H\times W}
    \]
    \item Convolution weights:
    \[
        W\in\mathbb{R}^{C_{\mathrm{out}}\times C_{\mathrm{in}}\times K_H\times K_W}
    \]
    \item Attention scores:
    \[
        A\in\mathbb{R}^{B\times H_{\mathrm{head}}\times L\times L}
    \]
\end{itemize}
\end{frame}

\begin{frame}{Convolution và attention là tensor contractions}
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
        \textbf{Convolution}
        \[
            Y_{b,o,h,w}
            =
            \sum_{c,u,v}
            W_{o,c,u,v}X_{b,c,h+u,w+v}
        \]
        \begin{itemize}
            \item Co theo mode channel và kernel.
            \item Low-rank conv thay một phép lớn bằng nhiều phép nhỏ.
        \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
        \textbf{Self-attention}
        \[
            A_{b,h,i,j}
            =
            \sum_k Q_{b,h,i,k}K_{b,h,j,k}
        \]
        \[
            O_{b,h,i,\delta}
            =
            \sum_j A_{b,h,i,j}V_{b,h,j,\delta}
        \]
        \begin{itemize}
            \item Tensor attention tăng theo $O(L^2)$.
        \end{itemize}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Polynomial networks / Pi-nets}
\[
    G(x)=\beta+\mathcal{W}^{[1]}\times_2x+
    \mathcal{W}^{[2]}\times_2x\times_3x+\cdots
\]
\begin{itemize}
    \item $\mathcal{W}^{[n]}$ mã hóa tương tác đặc trưng bậc $n$.
    \item Tensor đầy đủ có $O(od^n)$ tham số, tăng rất nhanh theo bậc.
    \item CP decomposition cho weight tensor:
    \[
        \mathcal{W}^{[n]}
        \approx
        \sum_{r=1}^{R}a^{(0)}_r\circ a^{(1)}_r\circ\cdots\circ a^{(n)}_r
    \]
    \item Số tham số giảm xấp xỉ còn $O(R(o+nd))$.
\end{itemize}
\end{frame}

\begin{frame}{Mixture of Experts dưới góc nhìn tensor}
\[
    y=\sum_{n=1}^{N}a_n(x)W_n^\top x,
    \qquad
    \mathcal{W}\in\mathbb{R}^{N\times I\times O}
\]
\[
    \mathcal{W}_{n,i,o}
    \approx
    \sum_{r=1}^{R}S_{n,r}U_{i,r}V_{o,r}
\]
\begin{itemize}
    \item Mode đầu tiên là expert index.
    \item $S$ điều khiển expert specialization.
    \item $U,V$ chia sẻ cấu trúc input/output giữa các expert.
\end{itemize}
\end{frame}

\begin{frame}{LoRTA và TensorGRAD}
\textbf{LoRTA: tensor hóa update LoRA}
\[
    \Delta\mathcal{W}_{\ell,h,o,i}
    \approx
    \sum_{r=1}^{R}
    A_{\ell,r}B_{h,r}C_{o,r}D_{i,r}
\]
\begin{itemize}
    \item Chia sẻ cấu trúc adaptation qua layer, head, output và input.
    \item Mở rộng từ low-rank matrix sang low-rank tensor.
\end{itemize}
\vspace{0.2cm}
\textbf{TensorGRAD: nén gradient}
\[
    \mathcal{G}_{\nabla}
    \approx
    \mathcal{C}\times_1U^{(1)}\times_2\cdots\times_NU^{(N)}
\]
\end{frame}

\begin{frame}{Poly-SA và attention hiệu quả}
\[
    f_{\mathrm{Poly\mbox{-}SA}}(X)
    =
    \Psi\big((XW_1)\ast(XW_2)\big)\ast XW_3
\]
\begin{itemize}
    \item Thay một phần softmax attention bằng cấu trúc đa thức.
    \item Mục tiêu: giảm chi phí attention từ bậc hai theo token về gần tuyến tính.
    \item Thông điệp chung: tensor product không chỉ để nén, mà còn gợi ý kiến trúc mới.
\end{itemize}
\end{frame}

\begin{frame}{Hạn chế của tensor hóa trong học sâu}
\begin{itemize}
    \item Rank quá thấp có thể làm mô hình underfit.
    \item Rank quá cao làm mất lợi thế nén.
    \item Tensor hóa không tự động tạo interpretability nếu các mode ban đầu không có ý nghĩa rõ.
    \item Cần chọn rank, cấu trúc factorization và vị trí áp dụng theo bài toán.
\end{itemize}
\end{frame}
