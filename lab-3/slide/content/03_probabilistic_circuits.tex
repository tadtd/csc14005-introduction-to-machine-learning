\section{Probabilistic Circuits}

\begin{frame}{Tensor xác suất và động lực}
\[
    \mathcal{P}\in
    \mathbb{R}^{|\mathcal{D}_1|\times\cdots\times|\mathcal{D}_N|},
    \qquad
    |\mathcal{P}|=\prod_{n=1}^{N}|\mathcal{D}_n|
\]
\begin{itemize}
    \item Mỗi entry là một xác suất $p(x_1,\ldots,x_N)$ của một assignment.
    \item Tensor xác suất đầy đủ tăng theo hàm mũ theo số biến, nên không thể materialize khi $N$ lớn.
    \item Tensor factorizations biểu diễn gọn; circuits tổ chức phép tính để trả lời truy vấn hiệu quả.
\end{itemize}
\[
    \underbrace{\text{Tensor Factorization}}_{\text{biểu diễn gọn}}
    \Longleftrightarrow
    \underbrace{\text{Circuit}}_{\text{tổ chức phép tính}}
    \Longrightarrow
    \underbrace{\text{Tractable Inference}}_{\text{suy luận hiệu quả}}
\]
\end{frame}

\begin{frame}{Circuit gồm những unit nào?}
\small
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.32\textwidth}
        \textbf{Input unit}
        \[
            c_n(x_{S_n})=f_n(x_{S_n})
        \]
        \begin{itemize}
            \item Đọc biến hoặc hàm cục bộ.
            \item Scope nhỏ: $S_n\subseteq\mathbf{X}$.
        \end{itemize}
    \end{column}
    \begin{column}{0.32\textwidth}
        \textbf{Product unit}
        \[
            c_n=\prod_{j\in\mathrm{in}(n)}c_j
        \]
        \begin{itemize}
            \item Ghép các scope con.
            \item Tractable khi decomposable.
        \end{itemize}
    \end{column}
    \begin{column}{0.32\textwidth}
        \textbf{Sum unit}
        \[
            c_n=\sum_{j\in\mathrm{in}(n)}w_{n,j}c_j
        \]
        \begin{itemize}
            \item Mixture/tổng có trọng số.
            \item Tractable khi smooth.
        \end{itemize}
    \end{column}
\end{columns}
\[
    \mathrm{scope}(n)=\bigcup_{j\in\mathrm{in}(n)}\mathrm{scope}(j)
\]
\end{frame}

\begin{frame}{Từ tensor factorization đến circuit}
\small
\begin{itemize}
    \item Quan sát chính trong report/tutorial: một circuit có thể tính một tensor entry tại một index bất kỳ.
    \[
        \mathcal{T}_{x_1,\ldots,x_N}\approx c(x_1,\ldots,x_N).
    \]
    \item Hướng TF $\to$ circuit: công thức factorization trở thành computational graph.
    \item Hướng circuit $\to$ TF: smooth và decomposable circuit mã hóa một multilinear polynomial, tương ứng với một tensor factorization.
\end{itemize}
\[
    \text{factor matrices/core}
    \quad\longrightarrow\quad
    \text{input/product/sum layers}
    \quad\longrightarrow\quad
    \text{query without full tensor}
\]
\end{frame}

\begin{frame}{Tucker decomposition như circuit}
\[
    \mathcal{T}_{x_1,\ldots,x_N}
    \approx
    \sum_{r_1,\ldots,r_N}
    \mathcal{W}_{r_1,\ldots,r_N}
    \prod_{n=1}^{N}V^{(n)}_{x_n,r_n}
\]
\begin{itemize}
    \item Factor matrices cung cấp input layers, tương tự CP.
    \item Core tensor $\mathcal{W}$ trộn các latent components $(r_1,\ldots,r_N)$.
    \item Circuit tương ứng có kích thước:
    \[
        |c|\in O\!\left(N\prod_{n=1}^{N}R_n\right).
    \]
    \item Khi core superdiagonal, Tucker suy biến thành CP.
\end{itemize}
\end{frame}

\begin{frame}{Tensor hóa circuit và layer rules}
\small
\begin{itemize}
    \item Tensor hóa circuit là cách xem forward pass của circuit như công thức tensor factorization.
    \item Report dùng ba quy tắc để xây circuit theo layer:
\end{itemize}
\[
    \mathbf{c}(\mathbf{x})=\mathbf{W}\boldsymbol{\ell}(\mathbf{x})
    \qquad
    \mathbf{c}=\boldsymbol{\ell}\had\mathbf{r}
    \qquad
    \mathbf{c}=\operatorname{vec}\!\left(\boldsymbol{\ell}\mathbf{r}^{\top}\right)
\]
\begin{itemize}
    \item Linear projection tương ứng sum/mixture layer.
    \item Hadamard product tạo CP-style product layer.
    \item Kronecker product tạo Tucker-style product layer trước khi chiếu tuyến tính.
\end{itemize}
\end{frame}

\begin{frame}{Lego block approach}
\small
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
        \textbf{Mục tiêu}
        \begin{itemize}
            \item Thống nhất các tensor factorizations theo block/layer.
            \item Ghép các block để tạo factorization mới.
            \item Giữ structural properties để suy luận vẫn tractable.
        \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
        \textbf{Các block}
        \begin{itemize}
            \item CP layer: Hadamard product rồi sum theo rank.
            \item Tucker layer: Kronecker product rồi linear projection bằng core/weight.
            \item Monarch layer: dùng structured weight matrix trong pipeline factorization.
        \end{itemize}
    \end{column}
\end{columns}
\[
    \text{input functions}
    \rightarrow
    \text{product block}
    \rightarrow
    \text{linear/sum block}
    \rightarrow
    \text{new circuit layer}
\]
\end{frame}

\begin{frame}{Deep circuit và hierarchical TF}
\small
\begin{itemize}
    \item Deep circuit thu được khi stack nhiều circuit layers theo ba quy tắc: tractable functions, linear projection, product.
    \item Deep circuit tương ứng với hierarchical tensor factorization.
    \item Ý nghĩa chính: chia nhỏ scope, chia sẻ sub-factorizations, biểu diễn một số hàm gọn hơn circuit nông.
    \item Độ gọn không tự đảm bảo: hiệu quả phụ thuộc region graph, layer type, rank/width và cấu trúc phụ thuộc thật trong dữ liệu.
\end{itemize}
\[
    \text{shallow CP/Tucker}
    \quad\longrightarrow\quad
    \text{stacked blocks}
    \quad\longrightarrow\quad
    \text{hierarchical/deep circuit}
\]
\end{frame}

\begin{frame}{Các truy vấn suy luận trong report}
\small

\toprule
\textbf{Query} & \textbf{Dạng toán học} & \textbf{Thao tác trên TF/circuit} \\
\midrule
Marginal & $\sum_{\mathbf{x}_M}p(\mathbf{x}_O,\mathbf{x}_M)$ & Sum-out/contract modes trong $M$ \\

Expectation & $\sum_{\mathbf{x}}p(\mathbf{x})f(\mathbf{x})$ & Product circuit hoặc tensor contraction \\
Fairness & $\mathbb{E}[f\mid A=0]-\mathbb{E}[f\mid A=1]$ & Chọn slice theo sensitive mode \\
Robustness & $\mathbb{E}_{\varepsilon}[f(\mathbf{x}+\varepsilon)]$ & Contract nếu compatible; nếu không thì sampling \\
\bottomrule
\end{tabular}

\begin{frame}{Marginal và conditional dưới góc nhìn TF}
\[
    c(x_1,\ldots,x_N)
    =
    \sum_{r=1}^{R}\prod_{n=1}^{N}V^{(n)}_{x_n,r}
\]
\[
    c(\mathbf{x}_O)
    =
    \sum_{r=1}^{R}
    \left(\prod_{n\in O}V^{(n)}_{x_n,r}\right)
    \left(\prod_{m\in M}\sum_{x_m\in\mathcal{D}_m}V^{(m)}_{x_m,r}\right)
\]
\begin{itemize}
    \item Marginalize mode $m$ bằng cách thay factor $V^{(m)}_{x_m,r}$ bởi tổng theo hàng.
    \item Conditional query dùng tử số là evaluate tại assignment, mẫu số là marginal trên biến missing.
\end{itemize}
\end{frame}

\begin{frame}{Expectation và reasoning về ML models}
\[
    \mathbb{E}_{\mathbf{x}\sim p}[f(\mathbf{x})]
    =
    \sum_{\mathbf{x}}p(\mathbf{x})f(\mathbf{x})
    =
    \langle\mathcal{P},\mathcal{F}\rangle
\]
\begin{itemize}
    \item Nếu $p$ và $f$ là compatible circuits, tính expectation bằng product circuit rồi marginalize.
    \item Dưới góc nhìn TF: contract tensor xác suất $\mathcal{P}$ với tensor hàm $\mathcal{F}$.
    \item Các ví dụ trong tutorial/report:
    \[
        \mathbb{E}[f(\mathbf{x})\mid A=0]
        -
        \mathbb{E}[f(\mathbf{x})\mid A=1],
        \qquad
        \mathbb{E}_{\varepsilon\sim p_{\mathrm{noise}}}[f(\mathbf{x}+\varepsilon)].
    \]
\end{itemize}
\end{frame}

\begin{frame}{Latent variable view, MAP và sampling}
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
        \textbf{Latent variable view}
        \[
            p(\mathbf{x})
            =
            \sum_{r=1}^{R}p(r)\prod_{n=1}^{N}p_n(x_n\mid r)
        \]
        \begin{itemize}
            \item Rank index $r$ là latent component.
            \item Sum node là mixture variable.
            \item Product node sinh độc lập trên scope rời nhau.
        \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
        \textbf{MAP / sampling}
        \begin{itemize}
            \item MAP/MPE cần thêm determinism để tối ưu tractable hơn.
            \item Sampling top-down: sum node chọn nhánh categorical theo weight.
            \item Product node đi qua tất cả nhánh con vì scope rời nhau.
            \item Input node lấy mẫu từ phân phối lá.
        \end{itemize}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Ứng dụng, mở rộng và câu hỏi mở}
\small
\begin{itemize}
    \item \textbf{Tractable trong pipeline không tractable}: dùng PC ở chỗ cần marginalization/expectation chính xác, rồi nối với neural encoder/decoder.
    \item \textbf{Từ rời rạc sang liên tục}: input functions có thể là Gaussian, spline, polynomial; tensor xác suất trở thành quasi-tensor/infinite-dimensional.
    \item \textbf{Hạn chế}: tractable inference không đồng nghĩa với biểu diễn luôn gọn; Tucker core có thể tăng như $R^N$.
    \item \textbf{Phụ thuộc cấu trúc}: hiệu quả biểu diễn phụ thuộc region graph, layer type, rank/width và dependency thật trong dữ liệu.
    \item \textbf{Câu hỏi mở}: optimization, identifiability, approximate inference, scalability.
\end{itemize}
