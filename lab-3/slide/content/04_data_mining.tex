\begin{frame}{CP-ALS: thuật toán nền tảng cho pattern discovery}
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.50\textwidth}
        \[
            \min_{A^{(1)},\ldots,A^{(N)}}
            \left\|\mathcal{X}-\cp{A^{(1)},\ldots,A^{(N)}}\right\|_F^2
        \]
        \[
            A^{(n)} \leftarrow X_{(n)}Z^{(n)}
            \left(Z^{(n)T}Z^{(n)}\right)^\dagger
        \]
        \[
            Z^{(n)} =
            A^{(N)}\odot\cdots\odot A^{(n+1)}
            \odot A^{(n-1)}\odot\cdots\odot A^{(1)}
        \]
    \end{column}
    \begin{column}{0.46\textwidth}
        \begin{itemize}
            \item Cố định các factor còn lại, cập nhật từng mode bằng least squares.
            \item $X_{(n)}Z^{(n)}$ là MTTKRP, thường là nút thắt tính toán chính.
            \item Gram matrix tính nhanh bằng Hadamard product:
            \[
                \Gamma^{(n)}=\mathop{\ast}_{k\ne n} A^{(k)T}A^{(k)}
            \]
        \end{itemize}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{CP-OPT, dữ liệu khuyết và loss khác}
\small

\begin{tabular}{p{0.22\textwidth}p{0.36\textwidth}p{0.32\textwidth}}
\toprule
\textbf{Biến thể} & \textbf{Công thức chính} & \textbf{Khi dùng} \\
\midrule
CP-OPT &
$\nabla_{A^{(n)}} f=A^{(n)}\Gamma^{(n)}-M^{(n)}$ &
Tối ưu tất cả factor cùng lúc bằng L-BFGS/gradient-based optimizer. \\
CP-WOPT &
$\|\mathcal{W}\ast(\mathcal{X}-\hat{\mathcal{X}})\|_F^2$ &
Chỉ fit trên entries quan sát được khi dữ liệu thiếu. \\
GCP-OPT &
$\sum_i \ell(X_i,\hat{X}_i)$ &
Thay Frobenius loss bằng loss phù hợp count/binary data. \\
\bottomrule
\end{tabular}

\vspace{0.25cm}
\[
    \widetilde{\mathcal{X}}
    =
    \mathcal{W}\ast\mathcal{X}
    +
    (1-\mathcal{W})\ast\hat{\mathcal{X}}
\]
\centering
EM-style imputation: quan sát giữ nguyên, missing entries lấy từ mô hình hiện tại.
\end{frame}
