\begin{frame}{CP-ALS: thuật toán nền tảng cho pattern discovery}
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.50\textwidth}
        \[
            \min_{A^{(1)},\ldots,A^{(N)}}
            \left\|\mathcal{X}-\cp{A^{(1)},\ldots,A^{(N)}}\right\|_F^2
        \]
        \[
            A^{(n)} \leftarrow X_{(n)}Z^{(n)}
            \left(Z^{(n)T}Z^{(n)}\right)^\dagger
        \]
        \[
            Z^{(n)} =
            A^{(N)}\odot\cdots\odot A^{(n+1)}
            \odot A^{(n-1)}\odot\cdots\odot A^{(1)}
        \]
    \end{column}
    \begin{column}{0.46\textwidth}
        \begin{itemize}
            \item Cố định các factor còn lại, cập nhật từng mode bằng least squares.
            \item $X_{(n)}Z^{(n)}$ là MTTKRP, thường là nút thắt tính toán chính.
            \item Gram matrix tính nhanh bằng Hadamard product:
            \[
                \Gamma^{(n)}=\mathop{\ast}_{k\ne n} A^{(k)T}A^{(k)}
            \]
        \end{itemize}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Ứng dụng, mở rộng và câu hỏi mở}
\begin{itemize}
    \item \textbf{Ứng dụng đa lĩnh vực:} EEG/fMRI, hóa sinh, hệ vi sinh; giữ ngữ nghĩa vật lý của từng mode.
    \item \textbf{Từ đa tuyến tính sang linh hoạt:} PARAFAC2 cho dữ liệu không đồng nhất; CMTF ghép nối dữ liệu đa nguồn qua coupling.
    \item \textbf{Hạn chế:} uniqueness và interpretability không miễn phí; nonlinearity cao làm factor khó diễn giải.
    \item \textbf{Phụ thuộc cấu trúc:} hiệu quả phụ thuộc loại phân rã, loss function, coupling và giả định dữ liệu.
    \item \textbf{Câu hỏi mở:} rank selection, prior knowledge, temporal misalignment, interpretability vs. nonlinearity.
\end{itemize}
\end{frame}
