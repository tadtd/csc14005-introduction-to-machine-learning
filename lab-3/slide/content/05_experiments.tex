\section{Thực nghiệm}

\begin{frame}{Demo: LoRA rank sweep}
\large
\begin{itemize}
    \item Task: GLUE SST-2 sentiment classification.
    \item Model: \texttt{prajjwal1/bert-tiny}.
    \item Baselines: frozen encoder, full fine-tuning.
    \item LoRA ranks: $r\in\{2,4,8\}$.
    \item Metrics: accuracy, loss, trainable parameters, runtime.
\end{itemize}
\end{frame}

\begin{frame}{Kết quả chính}
\centering
\small
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lrrrr}
\toprule
\textbf{Config} & \textbf{Rank} & \textbf{Trainable params} & \textbf{Best acc.} & \textbf{Time (s)} \\
\midrule
Frozen & -- & 258 & 0.6525 & 16.68 \\
Full fine-tune & -- & 4,386,178 & 0.7569 & 56.85 \\
LoRA & 2 & 2,306 & 0.7248 & 29.82 \\
LoRA & 4 & 4,354 & 0.7351 & 29.78 \\
LoRA & 8 & 8,450 & 0.7511 & 30.90 \\
\bottomrule
\end{tabular}
\end{frame}

\begin{frame}{Thông điệp từ thực nghiệm}
\large
\begin{itemize}
    \item LoRA rank 8 đạt $0.7511$ accuracy so với $0.7569$ của full fine-tuning.
    \item Số tham số trainable giảm từ $4{,}386{,}178$ xuống $8{,}450$.
    \item Đây là minh họa trực tiếp cho lợi ích của low-rank/tensor factorization trong học sâu.
\end{itemize}
\end{frame}
