\begin{frame}{LoRA rank sweep: nhận xét}
\begin{itemize}
    \item LoRA rank 8 chỉ kém full fine-tuning khoảng $0.57$ điểm accuracy.
    \item Số tham số trainable giảm từ $4{,}386{,}178$ xuống $8{,}450$.
    \item Best accuracy tăng khi rank tăng từ 2 lên 8 trong lần chạy này.
    \item Kết quả không nên xem là benchmark đầy đủ: model nhỏ, subset nhỏ, chỉ 3 epochs.
\end{itemize}
\[
    \text{LoRA r=8: }0.1923\%\text{ tham số trainable}
    \quad\text{và}\quad
    \approx 519\times\text{ ít tham số hơn full fine-tune}
\]
\end{frame}

\begin{frame}{CPJ Circuit: toy 4-bit}
\begin{itemize}
    \item Phân phối mục tiêu trên 4 biến nhị phân: $2^4=16$ entries.
    \item Model: Binary CPJ Circuit, width $K=8$.
    \item Train bằng maximum likelihood với Adam trong 300 epochs.
    \item Sau huấn luyện, enumerate toàn bộ 16 assignments để kiểm tra tensor xác suất.
\end{itemize}
\[
    \mathrm{KL}(\text{target}\Vert\text{circuit})
    =
    0.0054,
    \qquad
    \sum_x p_{\mathrm{circuit}}(x)=1.000000
\]
\end{frame}

\vspace{0.35cm}
\begin{itemize}
    \item Tensor synthetic kích thước $10\times15\times20$, noise $=0.1$.

\begin{frame}{Data Mining: PARAFAC2 và CMTF}
\begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
        \textbf{PARAFAC2}
        \begin{itemize}
            \item $K=5$ subjects, mỗi slice có số trial khác nhau.
            \item Rank $R=4$.
            \item Fit score hội tụ sau 47 vòng lặp, đạt $>0.7277$.
        \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
        \textbf{CMTF}
        \begin{itemize}
            \item Tensor EEG: $10\times64\times481$.
            \item Metadata giả lập: $10\times5$.
            \item Hai nguồn chia sẻ factor subject $A$.
        \end{itemize}
    \end{column}
\end{columns}
\[
    \min \|\mathcal{X}-\cp{A,B,C}\|_F^2+\|Y-AD^\top\|_F^2
\]
\end{frame}
